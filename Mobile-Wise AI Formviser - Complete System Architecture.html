Master Orchestration Flow

Level 1: Container Loading

1. User visits system → main-container.html loads

2. Creates 820×550px container with dual iframe architecture

3. Left iframe (200px): new-sidebar.html navigation

4. Right iframe (590px): voice-chat-module.html main app

1. voice-chat-module.html loads and shows splash screen

2. Displays microphone access image and "Activate Voice Chat" button

getAIResponse(message) {
    // 1. Check nci-practice-knowledge.js for context
    const context = NPIKnowledge.getContext(message);
    
    // 2. Generate base response  
    const baseResponse = BusinessResponses.getResponse(message);
    
    // 3. Apply closing technique
    const closingTechnique = ClosingTechniques.selectBest(context);
    
    // 4. Handle potential objections
    const objectionPreemption = ObjectionHandlers.preempt(message);
    
    // 5. Generate compelling CTA
    const cta = CTAGenerators.create(context, closingTechnique);
    
    // 6. Combine into powerful response
    return combineResponse(baseResponse, closingTechnique, objectionPreemption, cta);
}
